table 1: datasource (metadata about datasource ROOT file and per run data)
columns: run(int),filelocation(string),version(string),totallumisection(int)

table 2: lumidata (per-lumisec lumi data) 1 row per lumisection
columns: run(int),lumisection(int),instant_lumi(float),instant_lumi_err(float),instant_lumi_quality(float),instant_et_lumi(float),instant_et_lumi_err(float),instant_et_lumi_quality(float),normalization_et(int),instant_occ1_lumi(float),instant_occ1_lumi_err(float),instant_occ1_lumi_quality(float),normalization_occ1(int),instant_occ2_lumi(float),instant_occ2_lumi_err(float),instant_occ2_lumi_quality(float),normalization_occ2(int)

table 3: hltdata (per-lumisec hlt data) ~200 rows per lumisection
columns: run(int),lumisection(int),inputcount(int),acceptedcount(int),hltpath(string),prescale(int, same for all lumi sec in a run)

table 4: l1data (per-lumisec l1 data) 128+64 rows per lumisection
columns: run(int),lumisection(int),type(string: algo or tech),triggerbit(int),algorithm(string),count(int),deadtime(int),prescale(int, same for all lumi sec in a run)


reference doc: https://twiki.cern.ch/twiki/bin/view/CMS/LumiWiki_ROOTFileAnalysis
